**Goal:** Build a Streamlit-based HITL review interface that samples ~1.150 ClingoPredicates from both corpora, scores them with an alhedonic signal for priority ordering, and persists each SOVEREIGN/ELASTIC/SKIP decision incrementally to JSON with resume support.

**Architecture:** `sampler.py` loads `.lp` files into `HitlItem` dataclasses and applies the four-component alhedonic score; `checkpoint.py` manages the session state and writes each decision atomically; `app.py` drives the Streamlit UI reading from checkpoint state. The CLI (`__main__.py`) orchestrates the three actions: `sample`, `review`, `report`.

**Tech Stack:** Python 3.11, Streamlit ≥1.35, Pydantic v2 (schemas.py read-only), pytest, ruff, mypy --strict. Uses `SovereigntyClass` from `qfeng.core.schemas`.

```python
"""E4 — Stratified sampler + alhedonic scorer for HITL review."""
from __future__ import annotations

import re
from dataclasses import dataclass, field
from pathlib import Path


@dataclass
class HitlItem:
    """A single predicate ready for HITL review."""
    atom_id: str
    chunk_id: str
    strength: str
    rule: str
    modality: str
    corpus: str
    source_doc: str
    alhedonic_score: float = 0.0
    concurrent_chunks: list[str] = field(default_factory=list)


_COMMENT_RE = re.compile(
    r"%\s*atom_id:\s*(\S+)\s*\|\s*chunk:\s*(\S+)\s*\|\s*strength:\s*(\S+)"
)
_MODALITY_RE = re.compile(r"^(obligated|prohibited|permitted)\(")


def load_predicates_from_lp(
    lp_path: Path,
    corpus: str,
    concurrent_chunks: dict[str, list[str]],
) -> list[HitlItem]:
    """Parse a .lp file into HitlItems. Skips concurrent_facts.lp."""
    if lp_path.stem == "concurrent_facts":
        return []
    text = lp_path.read_text(encoding="utf-8", errors="replace")
    blocks = [b.strip() for b in text.strip().split("\n\n") if b.strip()]
    items: list[HitlItem] = []
    for block in blocks:
        m = _COMMENT_RE.search(block)
        if not m:
            continue
        atom_id, chunk_id, strength = m.group(1), m.group(2), m.group(3)
        rest = block.split("\n", 1)[1].strip() if "\n" in block else ""
        mod_m = _MODALITY_RE.match(rest)
        modality = mod_m.group(1) if mod_m else "unknown"
        items.append(HitlItem(
            atom_id=atom_id,
            chunk_id=chunk_id,
            strength=strength,
            rule=block,
            modality=modality,
            corpus=corpus,
            source_doc=lp_path.stem,
            concurrent_chunks=concurrent_chunks.get(chunk_id, []),
        ))
    return items


def score_alhedonic(
    item: HitlItem,
    all_items_by_chunk: dict[str, list[HitlItem]],
) -> float:
    """Placeholder — implemented in Task 3."""
    return 0.0


def sample_stratified(
    items: list[HitlItem],
    target: int,
    min_per_modality: int = 1,
) -> list[HitlItem]:
    """Placeholder — implemented in Task 4."""
    return items[:target]
```

```python
def _load_concurrent_map(lp_dir: Path) -> dict[str, list[str]]:
    """Load concurrent_facts.lp → {chunk_id: [chunk_id, ...]}."""
    conc_path = lp_dir / "concurrent_facts.lp"
    result: dict[str, list[str]] = {}
    if not conc_path.exists():
        return result
    pattern = re.compile(r"concurrent\((\w+),\s*(\w+)\)\.")
    for line in conc_path.read_text(encoding="utf-8").splitlines():
        m = pattern.search(line)
        if m:
            a, b = m.group(1), m.group(2)
            result.setdefault(a, []).append(b)
            result.setdefault(b, []).append(a)
    return result


def score_alhedonic(
    item: HitlItem,
    all_items_by_chunk: dict[str, list[HitlItem]],
) -> float:
    """Alhedonic priority score [0.0, 1.0].

    concurrent_penalty (0.4): chunk has concurrent peers
    modality_conflict  (0.3): concurrent peers have different modality
    strength_mismatch  (0.2): concurrent peers have different strength
    low_confidence     (0.1): strength == 'operational'
    """
    score = 0.0
    if item.concurrent_chunks:
        score += 0.4
        peer_modalities = {
            peer.modality
            for cid in item.concurrent_chunks
            for peer in all_items_by_chunk.get(cid, [])
        }
        if peer_modalities and item.modality not in peer_modalities:
            score += 0.3
        peer_strengths = {
            peer.strength
            for cid in item.concurrent_chunks
            for peer in all_items_by_chunk.get(cid, [])
        }
        if peer_strengths and item.strength not in peer_strengths:
            score += 0.2
    if item.strength == "operational":
        score += 0.1
    return min(score, 1.0)
```

```python
def sample_stratified(
    items: list[HitlItem],
    target: int,
    min_per_modality: int = 1,
) -> list[HitlItem]:
    """Stratified sample: proportional per doc, min per modality, signal-first order."""
    if not items:
        return []

    by_doc: dict[str, list[HitlItem]] = {}
    for item in items:
        by_doc.setdefault(item.source_doc, []).append(item)

    n_docs = len(by_doc)
    base_per_doc = max(5, target // max(n_docs, 1))

    selected: list[HitlItem] = []
    for doc in sorted(by_doc):
        doc_items = sorted(by_doc[doc], key=lambda x: x.alhedonic_score, reverse=True)
        # Guarantee min_per_modality coverage
        seen_mod: set[str] = set()
        guaranteed: list[HitlItem] = []
        rest: list[HitlItem] = []
        for it in doc_items:
            if it.modality not in seen_mod:
                guaranteed.append(it)
                seen_mod.add(it.modality)
            else:
                rest.append(it)
        quota = min(base_per_doc, len(doc_items))
        take = (guaranteed + rest)[:quota]
        selected.extend(take)

    # Final sort by alhedonic_score, trim to target
    selected.sort(key=lambda x: x.alhedonic_score, reverse=True)
    return selected[:target]
```

```python
# src/qfeng/c1_digestion/hitl/checkpoint.py
"""E4 — Incremental checkpoint: persist HITL decisions to JSON, resume support."""
from __future__ import annotations

import json
from dataclasses import asdict, dataclass
from datetime import datetime, timezone
from pathlib import Path
from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from qfeng.c1_digestion.hitl.sampler import HitlItem


@dataclass
class Decision:
    atom_id: str
    verdict: str        # SOVEREIGN | ELASTIC | SKIP
    corpus: str
    source_doc: str
    decided_at: str     # ISO 8601


@dataclass
class HitlSession:
    corpus: str
    sample_path: Path
    decisions_path: Path
    report_path: Path
    items: list[HitlItem]
    decisions: dict[str, Decision]


def load_session(
    corpus: str,
    session_dir: Path,
    items: list[HitlItem],
) -> HitlSession:
    """Load or create a HITL session. Resumes existing decisions."""
    session_dir.mkdir(parents=True, exist_ok=True)
    sample_path = session_dir / "sample.json"
    decisions_path = session_dir / "decisions.json"
    report_path = session_dir / "report.md"

    # Write sample once
    if not sample_path.exists():
        sample_path.write_text(
            json.dumps([asdict(it) for it in items], ensure_ascii=False, indent=2),
            encoding="utf-8",
        )

    # Load existing decisions
    decisions: dict[str, Decision] = {}
    if decisions_path.exists():
        raw = json.loads(decisions_path.read_text(encoding="utf-8"))
        for d in raw:
            decisions[d["atom_id"]] = Decision(**d)

    return HitlSession(
        corpus=corpus,
        sample_path=sample_path,
        decisions_path=decisions_path,
        report_path=report_path,
        items=items,
        decisions=decisions,
    )


def save_decision(
    session: HitlSession,
    atom_id: str,
    verdict: str,
    source_doc: str,
) -> None:
    """Persist a single decision immediately. Thread-safe for single-process Streamlit."""
    dec = Decision(
        atom_id=atom_id,
        verdict=verdict,
        corpus=session.corpus,
        source_doc=source_doc,
        decided_at=datetime.now(timezone.utc).isoformat(),
    )
    session.decisions[atom_id] = dec

    all_decisions = list(session.decisions.values())
    session.decisions_path.write_text(
        json.dumps([asdict(d) for d in all_decisions], ensure_ascii=False, indent=2),
        encoding="utf-8",
    )


def generate_report(session: HitlSession) -> str:
    """Generate and write report.md. Returns markdown string."""
    total = len(session.items)
    decided = len(session.decisions)
    sovereign = sum(1 for d in session.decisions.values() if d.verdict == "SOVEREIGN")
    elastic = sum(1 for d in session.decisions.values() if d.verdict == "ELASTIC")
    skip = sum(1 for d in session.decisions.values() if d.verdict == "SKIP")
    remaining = total - decided

    by_doc: dict[str, dict[str, int]] = {}
    for dec in session.decisions.values():
        by_doc.setdefault(dec.source_doc, {"SOVEREIGN": 0, "ELASTIC": 0, "SKIP": 0})
        by_doc[dec.source_doc][dec.verdict] += 1

    doc_rows = "\n".join(
        f"| {doc} | {counts['SOVEREIGN']} | {counts['ELASTIC']} | {counts['SKIP']} |"
        for doc, counts in sorted(by_doc.items())
    )

    report = f"""# E4 HITL Validation Report — {session.corpus}

## Resumo

| Métrica | Valor |
|---------|-------|
| Total na amostra | {total} |
| Revisados | {decided} |
| Restantes | {remaining} |
| SOVEREIGN | {sovereign} |
| ELASTIC | {elastic} |
| SKIP | {skip} |

## Por Documento

| Documento | SOVEREIGN | ELASTIC | SKIP |
|-----------|-----------|---------|------|
{doc_rows}
"""
    session.report_path.write_text(report, encoding="utf-8")
    return report
```

```python
CORPUS_CONFIG: dict[str, dict[str, object]] = {
    "sus_validacao": {
        "e3_dir": Path("outputs/e3_predicates"),
        "session_dir": Path("outputs/hitl_session/sus_validacao"),
        "target": 990,
    },
    "advocacia_trabalhista": {
        "e3_dir": Path("outputs/e3_predicates_trabalhista"),
        "session_dir": Path("outputs/hitl_session/advocacia_trabalhista"),
        "target": 160,
    },

def _load_all_items(e3_dir: Path, corpus: str) -> list[HitlItem]:
    """Load all HitlItems from e3_dir, scoring each with alhedonic signal."""
    concurrent_map = _load_concurrent_map(e3_dir)
    # Also check subdirectories (e.g. brasil/, brasil_trabalhista/)
    lp_files: list[Path] = list(e3_dir.rglob("*.lp"))
    lp_files = [f for f in lp_files if f.stem != "concurrent_facts"]

    all_items: list[HitlItem] = []
    for lp in sorted(lp_files):
        all_items.extend(load_predicates_from_lp(lp, corpus=corpus, concurrent_chunks=concurrent_map))

    # Build lookup for scoring
    by_chunk: dict[str, list[HitlItem]] = {}
    for item in all_items:
        by_chunk.setdefault(item.chunk_id, []).append(item)

    for item in all_items:
        item.alhedonic_score = score_alhedonic(item, by_chunk)

    return all_items


def cmd_sample(corpus: str) -> None:
    cfg = CORPUS_CONFIG[corpus]
    e3_dir = Path(str(cfg["e3_dir"]))
    session_dir = Path(str(cfg["session_dir"]))
    target = int(str(cfg["target"]))

    all_items = _load_all_items(e3_dir, corpus)
    logger.info("Carregados %d predicados de %s", len(all_items), e3_dir)

    sample = sample_stratified(all_items, target=target, min_per_modality=1)
    logger.info("Amostra: %d predicados (target=%d)", len(sample), target)

    session = load_session(corpus, session_dir, sample)
    already = len(session.decisions)
    pending = len(sample) - already
    logger.info("Sessão: %d já revisados, %d pendentes", already, pending)
    logger.info("sample.json → %s", session.sample_path)

    high_signal = sum(1 for it in sample if it.alhedonic_score > 0.5)
    logger.info("Predicados com signal > 0.5 (destacados): %d", high_signal)


def cmd_review(corpus: str) -> None:
    cfg = CORPUS_CONFIG[corpus]
    session_dir = Path(str(cfg["session_dir"]))
    app_path = Path(__file__).parent / "app.py"
    env_args = ["--", f"--corpus={corpus}", f"--session-dir={session_dir}"]
    cmd = [sys.executable, "-m", "streamlit", "run", str(app_path)] + env_args
    logger.info("Iniciando Streamlit: %s", " ".join(cmd))
    subprocess.run(cmd, check=True)


def cmd_report(corpus: str) -> None:
    cfg = CORPUS_CONFIG[corpus]
    session_dir = Path(str(cfg["session_dir"]))
    e3_dir = Path(str(cfg["e3_dir"]))
    target = int(str(cfg["target"]))

    all_items = _load_all_items(e3_dir, corpus)
    sample = sample_stratified(all_items, target=target, min_per_modality=1)
    session = load_session(corpus, session_dir, sample)
    report = generate_report(session)
    print(report)
    logger.info("Relatório → %s", session.report_path)


def main() -> None:
    parser = argparse.ArgumentParser(description="E4 HITL Validation CLI")
    parser.add_argument(
        "--action", required=True, choices=["sample", "review", "report"],
        help="Ação a executar",
    )
    parser.add_argument(
        "--corpus", required=True, choices=list(CORPUS_CONFIG.keys()),
        help="Corpus alvo",
    )
    args = parser.parse_args()

    if args.action == "sample":
        cmd_sample(args.corpus)
    elif args.action == "review":
        cmd_review(args.corpus)
    elif args.action == "report":
        cmd_report(args.corpus)


main()
```
